python database_setup.py

python data_setup.py

python application.py

# Explaination

addItem.html - add item layout
categoryItem.html - show item list by category layout
deleteItem.html - delete item layout
detailItem.html - show item detail layout
editItem.html - edit item layout
layout.html - default template
login.html - login layout
main.html - main layout

main.css - overall styling
